class JSON {
  <<interface>>
  +getConfig()
  +getByPath()

  <<interface>>

  +getStr()
  +getInt()

class MapWrapper {
  <<base>>
}

class JSONObject {
  -JSONConfig config
  +JSONObject(...)
  +set(key, value)
  +setOnce(key, value, filter)
  +putOpt(key, value)
  +accumulate(key, value)
  +append(key, value)
  +increment(key)
  +getByPath(expr)
  +toJSONString(indent, filter)
  +write(writer, indentFactor, indent, filter)
}

class JSONArray {
  -List rawList
  -JSONConfig config
  +JSONArray(...)
  +set(value)
  +add(index, value)
}

class JSONConfig {
  +ignoreNullValue
  +ignoreCase
  +dateFormat
  +checkDuplicate
  +ignoreError
}

class JSONUtil {
  <<facade>>
  +parseObj(...)
  +parseArray(...)
  +parse(...)
  +toJsonStr(...)
  +toBean(...)
  +parseFromXml(...)
  +toXmlStr(...)
  +wrap(...)
}

class ObjectMapper {
  <<mapper>>
  +of(source)
  +map(JSONObject, filter)
  +map(JSONArray, filter)

  <<parser>>

  +nextString()

  <<serializer>>
  +beginObj()

  +writeValue(value)

class JSONConverter {
  <<converter>>

  +jsonToBean(type, value, ignoreError)

  <<internal_util>>

  <<interface>>
  +parse(value)
