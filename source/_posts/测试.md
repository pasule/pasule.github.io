classDiagram

class JSON {
  <<interface>>
  +getConfig()
  +getByPath()
  +putByPath()
  +toJSONString()
  +write()
}

class JSONGetter {
  <<interface>>
  +getObj()
  +getStr()
  +getInt()
  +getJSONArray()
  +getJSONObject()
}

class MapWrapper~String,Object~ {
  <<base>>
}

class JSONObject {
  -JSONConfig config
  +JSONObject(...)
  +set(key, value)
  +setOnce(key, value, filter)
  +putOpt(key, value)
  +accumulate(key, value)
  +append(key, value)
  +increment(key)
  +getByPath(expr)
  +toJSONString(indent, filter)
  +write(writer, indentFactor, indent, filter)
}

class JSONArray {
  -List~Object~ rawList
  -JSONConfig config
  +JSONArray(...)
  +set(value)
  +add(index, value)
}

class JSONConfig {
  +ignoreNullValue
  +ignoreCase
  +dateFormat
  +checkDuplicate
  +ignoreError
}

class JSONUtil {
  <<facade>>
  +parseObj(...)
  +parseArray(...)
  +parse(...)
  +toJsonStr(...)
  +toBean(...)
  +parseFromXml(...)
  +toXmlStr(...)
  +wrap(...)
}

class ObjectMapper {
  <<mapper>>
  +of(source)
  +map(JSONObject, filter)
  +map(JSONArray, filter)
}

class JSONTokener {
  <<parser>>
  +nextValue()
  +nextString()
  +nextClean()
}

class JSONWriter {
  <<serializer>>
  +beginObj()
  +writeField(pair, filter)
  +writeValue(value)
  +end()
}

class JSONConverter {
  <<converter>>
  +jsonConvert(type, value, config)
  +jsonToBean(type, value, ignoreError)
}

class InternalJSONUtil {
  <<internal util>>
  +createRawMap(capacity, config)
  +testValidity(obj)
  +defaultIgnoreNullValue(source)
  +toCopyOptions(config)
}

class JSONBeanParser {
  <<interface>>
  +parse(value)
}

class JSONSupport {
  +parse(json)
  +toJSON()
  +toJSONString()
}

class XML {
  <<facade>>
  +toJSONObject(xml)
  +toXml(obj)
}

class JSONXMLParser {
  <<xml parser>>
}

class JSONXMLSerializer {
  <<xml serializer>>
}

JSONObject --|> MapWrapper

JSONArray ..|> JSON

JSONObject --> JSONConfig : 持有配置
JSONArray --> JSONConfig : 持有配置

JSONUtil --> JSONArray : 创建/返回
JSONUtil --> JSONConverter : toBean
JSONUtil --> XML : XML互转
JSONUtil --> JSONTokener : 间接触发解析
JSONUtil --> JSONWriter : 间接触发输出

JSONObject --> ObjectMapper : 构造时映射 source
JSONObject --> InternalJSONUtil : 创建Map/值校验
JSONObject --> JSONUtil : wrap(value)
JSONObject --> JSONArray : accumulate/append

JSONObject --> JSONConverter : getByPath(..., type)

ObjectMapper --> JSONTokener : 字符串/流解析
ObjectMapper --> InternalJSONUtil : CopyOptions/属性填充
ObjectMapper --> JSONObject : 填充对象
ObjectMapper --> JSONArray : 填充数组

JSONConverter --> JSONBeanParser : 自定义反序列化
JSONSupport ..|> JSONBeanParser
JSONSupport --> JSONConverter : parse时转Bean
